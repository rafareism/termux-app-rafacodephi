L("RAF_HOT RAF_NOINLINE void raf_memset(void* dst, u8 val, usize n) {")
L("    u8* p = (u8*)dst;")
L("#ifdef RAF_HAS_NEON")
L("    if (n >= 64 && ((uptr)p & 0xF) == 0) {")
L("        /* Replica byte em 16 posições via NEON — 64 bytes por iteração */")
L("        u8 tmp[16] __attribute__((aligned(16)));")
L("        for (u32 i = 0; i < 16; i++) tmp[i] = val;")
L("        __asm__ __volatile__(")
L('            "ld1 {v0.16b}, [%0]\\n\\t"')
L("            :: \"r\"(tmp) : \"v0\"")
L("        );")
L("        while (n >= 64) {")
L("            __asm__ __volatile__(")
L('                "st1 {v0.16b}, [%0], #16\\n\\t"')
L('                "st1 {v0.16b}, [%0], #16\\n\\t"')
L('                "st1 {v0.16b}, [%0], #16\\n\\t"')
L('                "st1 {v0.16b}, [%0], #16\\n\\t"')
L("                : \"+r\"(p) :: \"memory\"")
L("            );")
L("            n -= 64;")
L("        }")
L("    }")
L("    while (n >= 16 && ((uptr)p & 0xF) == 0) {")
L("        __asm__ __volatile__(")

L("        );")
L("        n -= 16;")
L("    }")
L("#else")
L("    /* fallback: 4 bytes por vez alinhado */")

L("/* CRC32C de 4 bytes em paralelo (ARM64 CRC32CW) */")

L("/* FNV-1a 64-bit — determinístico por spec */")

L("/* BLAKE3-simplified: hash de 32 bytes em BSS — sem heap */")

L("        0x6A09E667u, 0xBB67AE85u, 0x3C6EF372u, 0xA54FF53Au,")
L("        0x510E527Fu, 0x9B05688Cu, 0x1F83D9ABu, 0x5BE0CD19u")
L("    };")

L("    raf_memcpy(state, IV, 32);")
L("    u64 h = raf_hash64_fnv1a(data, len);")

L(" * Grupos de 4 bytes little-endian, XOR final = 0xF8F8DF32 */")

L("    u32 acc = 0;")
L("    usize i = 0;")
L("    while (i + 4 <= len) {")
L("        u32 w = (u32)data[i]")
L("              | ((u32)data[i+1] << 8)")
L("              | ((u32)data[i+2] << 16)")
L("              | ((u32)data[i+3] << 24);")

L("    }")
L("    /* padding com 0x00 no último grupo incompleto */")
L("    if (i < len) {")
L("        u32 w = 0;")
L("        for (usize j = 0; i+j < len; j++) w |= ((u32)data[i+j] << (j*8));")

L("    }")

L("/* Fast square root — Newton-Raphson")
L(" * Precisão: erro < 0.0001% — Ref: BOOSTERS.md §Raiz Quadrada Rápida */")
L("RAF_INLINE float fm_sqrt(float x) {")
L("    if (x <= 0.0f) return 0.0f;")
L("    union { float f; u32 i; } u;")
L("    u.f = x;")
L("    u.i = (u.i >> 1) + 0x1fbb4000u;  /* aproximação mágica */")
L("    u.f = 0.5f * (u.f + x / u.f);    /* Newton-Raphson iter 1 */")
L("    u.f = 0.5f * (u.f + x / u.f);    /* Newton-Raphson iter 2 */")
L("    return u.f;")
L("}")
L()
L("/* Fast inverse square root — Quake III 'fast InvSqrt'")
L(" * Ref: BOOSTERS.md §Raiz Quadrada Recíproca — 3.5x speedup */")
L("RAF_INLINE float fm_rsqrt(float x) {")
L("    union { float f; u32 i; } u;")
L("    u.f = x;")
L("    u.i = 0x5f3759dfu - (u.i >> 1);")
L("    u.f = u.f * (1.5f - 0.5f * x * u.f * u.f);")
L("    return u.f;")
L("}")
L()
L("/* Fast exp — Taylor series (1.9x speedup, 10 termos) */")
L("RAF_INLINE float fm_exp(float x) {")
L("    float result = 1.0f, term = 1.0f;")
L("    for (s32 i = 1; i <= 10; i++) { term *= x / (float)i; result += term; }")
L("    return result;")
L("}")

L("/* Fast log — manipulação de expoente IEEE-754 (2.2x speedup) */")
L("RAF_INLINE float fm_log(float x) {")
L("    if (x <= 0.0f) return -3.4028235e+38f; /* -FLT_MAX */")
L("    union { float f; u32 i; } u;")
L("    u.f = x;")
L("    s32 exp = (s32)((u.i >> 23) & 0xFFu) - 127;")
L("    u.i = (u.i & 0x007FFFFFu) | 0x3F800000u;")
L("    float m = u.f;")
L("    float log_m = (m - 1.0f) * (1.0f - 0.5f * (m - 1.0f));")
L("    return (float)exp * 0.693147f + log_m;")
L("}")

L("RAF_INLINE float fm_abs(float x)  { union { float f; u32 i; } u; u.f = x; u.i &= 0x7FFFFFFFu; return u.f; }")
L("RAF_INLINE float fm_min(float a, float b) { return a < b ? a : b; }")
L("RAF_INLINE float fm_max(float a, float b) { return a > b ? a : b; }")
L("RAF_INLINE float fm_clamp(float v, float lo, float hi) { return fm_min(fm_max(v, lo), hi); }")
L("RAF_INLINE float fm_lerp(float a, float b, float t) { return a + (b - a) * t; }")
L()
L("/* sin/cos aproximado — séries de Maclaurin (5 termos cada) */")
L("#define RAF_PI  3.14159265358979323846f")
L("#define RAF_PI2 6.28318530717958647692f")
L("RAF_INLINE float fm_sin_approx(float x) {")
L("    /* Normaliza para [-π, π] */")
L("    while (x >  RAF_PI) x -= RAF_PI2;")
L("    while (x < -RAF_PI) x += RAF_PI2;")
L("    float x2 = x*x;")
L("    return x * (1.0f - x2/6.0f * (1.0f - x2/20.0f * (1.0f - x2/42.0f)));")
L("}")
L("RAF_INLINE float fm_cos_approx(float x) { return fm_sin_approx(x + RAF_PI/2.0f); }")
L()
L("/* Float16 → Float32 sem libc — Ref: DeepSeek-RafCoder (Q4_0 baremetal) */")
L("RAF_INLINE float raf_f16_to_f32(u16 h) {")
L("    u32 sign     = (u32)(h >> 15);")
L("    u32 exponent = (u32)((h >> 10) & 0x1Fu);")
L("    u32 mantissa = (u32)(h & 0x3FFu);")
L("    u32 f32;")
L("    if (exponent == 0) {")
L("        if (mantissa == 0) { f32 = sign << 31; }")
L("        else {")
L("            exponent = 1;")
L("            while (!(mantissa & 0x400u)) { mantissa <<= 1; exponent--; }")
L("            mantissa &= 0x3FFu;")
L("            f32 = (sign<<31) | ((exponent+127-15)<<23) | (mantissa<<13);")
L("        }")
L("    } else if (exponent == 31) {")
L("        f32 = (sign<<31) | (0xFFu<<23) | (mantissa<<13);")
L("    } else {")
L("        f32 = (sign<<31) | ((exponent+127-15)<<23) | (mantissa<<13);")
L("    }")
L("    float result; raf_memcpy(&result, &f32, 4); return result;")
L("}")
L()
L("/* Float32 → Float16 */")
L("RAF_INLINE u16 raf_f32_to_f16(float v) {")
L("    u32 bits; raf_memcpy(&bits, &v, 4);")
L("    u32 sign     = (bits >> 31) & 1u;")
L("    u32 exponent = (bits >> 23) & 0xFFu;")
L("    u32 mantissa =  bits        & 0x7FFFFFu;")
L("    if (exponent == 0xFF) return (u16)((sign<<15)|(0x1Fu<<10)|(mantissa?1:0));")
L("    s32 e = (s32)exponent - 127 + 15;")
L("    if (e >= 31) return (u16)((sign<<15)|(0x1Fu<<10));")
L("    if (e <= 0)  return (u16)(sign<<15);")
L("    return (u16)((sign<<15)|((u32)e<<10)|(mantissa>>13));")
L("}")
L()

L("// SEÇÃO 7 — LOG CIRCULAR APPEND-ONLY (sem printf, sem fprintf, sem libc)")

L("#define RAF_LOG_MAGIC   0xA0F00001u")

L("#define RAF_LOG_WRN     2u")
L("#define RAF_LOG_ERR     3u")
L("#define RAF_LOG_FATAL   4u")
L()
L("typedef struct RAF_PACKED {")
L("    u32 magic;")
L("    u16 payload_len;")
L("    u8  level;")
L("    u8  flags;    /* bit0=overflow, bit1=truncated */")
L("    u32 seq;")

L("static u8  _g_logbuf[RAF_LOG_BUF_SIZE] RAF_ALIGN16 RAF_SECTION(\".bss\");")

L("    payload[pl++] = ']'; payload[pl++] = '[';")
L("    raf_memcpy(payload+pl, tag, tlen); pl += tlen;")

L("}")

# SEÇÃO 8 — BITRAF
L("// " + "=" * 117)
L("// SEÇÃO 8 — BITRAF: BIT-LEVEL OPERATIONS (nibble HI/LO, planos de bits, Witness)")
L("// Ref: llamaRafaelia/rafaelia-baremetal/src/bitraf.c")
L("//      llamaRafaelia/SPEC.md §BitStack World Model v1")
L("//      Vectras-VM-Android/engine/rmr/ (BITRAF sector invariant)")
L("//      termux-app-rafacodephi/RAFAELIA_MODULE_ORGANIZATION.md")
L("//")
L("// INVARIANTE FUNDAMENTAL do BitStack World Model v1:")
L("//   'Nenhuma computação consome bloco com Witness=false'")
L("//   Witness = bit7 do campo ctrl de cada RafBlock.")
L("//")
L("// Nibble HI = bits 7:4 de cada byte → plano de metadados")
L("// Nibble LO = bits 3:0 de cada byte → plano de dados")
L("// Geometria como índice → acesso O(1) sem hash")
L("// " + "=" * 117)
L()
L("typedef struct RAF_PACKED {")
L("    u8  data[59];   /* 59 bytes payload */")
L("    u8  crc_lo[4];  /* CRC32C dos 59 bytes (little-endian) */")
L("    u8  ctrl;       /* bit7=Witness, bit6=dirty, bits5:4=type, bits3:0=flags */")
L("} RafBlock;  /* exatamente 64 bytes */")
L()
L("/* Asserção estática sem assert.h */")
L("typedef char _raf_block_sz_check[(sizeof(RafBlock)==64)?1:-1];")
L()
L("#define RAF_BLOCK_WITNESS_GET(b)   (((b)->ctrl >> 7) & 1)")
L("#define RAF_BLOCK_WITNESS_SET(b)   ((b)->ctrl |=  0x80u)")
L("#define RAF_BLOCK_WITNESS_CLR(b)   ((b)->ctrl &= ~0x80u)")
L("#define RAF_BLOCK_DIRTY_GET(b)     (((b)->ctrl >> 6) & 1)")
L("#define RAF_BLOCK_DIRTY_SET(b)     ((b)->ctrl |=  0x40u)")
L("#define RAF_BLOCK_DIRTY_CLR(b)     ((b)->ctrl &= ~0x40u)")
L("#define RAF_BLOCK_TYPE_GET(b)      (((b)->ctrl >> 4) & 0x3u)")
L("#define RAF_BLOCK_TYPE_SET(b,t)    ((b)->ctrl = ((b)->ctrl & 0xCFu) | (((t)&3)<<4))")
L("#define RAF_BLOCK_FLAGS_GET(b)     ((b)->ctrl & 0xFu)")
L("#define RAF_BLOCK_FLAGS_SET(b,f)   ((b)->ctrl = ((b)->ctrl & 0xF0u) | ((f)&0xFu))")
L()
L("RAF_INLINE u8 raf_nibble_hi(u8 b)           { return (b >> 4) & 0xFu; }")
L("RAF_INLINE u8 raf_nibble_lo(u8 b)           { return  b       & 0xFu; }")
L("RAF_INLINE u8 raf_nibble_pack(u8 hi, u8 lo) { return ((hi & 0xFu) << 4) | (lo & 0xFu); }")
L()
L("static RafBlock _g_bitraf_blocks[RAF_MAX_BLOCKS] RAF_ALIGN64 RAF_SECTION(\".bss\");")
L("static u32      _g_bitraf_count  = 0;")
L("static u32      _g_bitraf_sealed = 0;")
L("static u32      _g_bitraf_errors = 0;")
L()
L("RAF_INLINE void raf_bitraf_init(void) {")
L("    raf_memset((void*)_g_bitraf_blocks, 0, sizeof(RafBlock)*RAF_MAX_BLOCKS);")
L("    _g_bitraf_count = _g_bitraf_sealed = _g_bitraf_errors = 0;")
L("}")
L()
L("RAF_INLINE s32 raf_bitraf_alloc_block(void) {")
L("    if (RAF_UNLIKELY(_g_bitraf_count >= RAF_MAX_BLOCKS)) return -1;")
L("    s32 idx = (s32)_g_bitraf_count++;")
L("    raf_memset(&_g_bitraf_blocks[idx], 0, sizeof(RafBlock));")
L("    RAF_BLOCK_WITNESS_CLR(&_g_bitraf_blocks[idx]);")
L("    return idx;")
L("}")
L()
L("RAF_INLINE void raf_bitraf_seal(s32 idx) {")
L("    if (RAF_UNLIKELY(idx < 0 || (u32)idx >= _g_bitraf_count)) return;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    u32 crc = raf_crc32c_fast(b->data, 59);")
L("    b->crc_lo[0] = (u8)(crc >>  0);")
L("    b->crc_lo[1] = (u8)(crc >>  8);")
L("    b->crc_lo[2] = (u8)(crc >> 16);")
L("    b->crc_lo[3] = (u8)(crc >> 24);")
L("    RAF_BLOCK_WITNESS_SET(b);")
L("    RAF_BLOCK_DIRTY_CLR(b);")
L("    _g_bitraf_sealed++;")
L("}")
L()
L("RAF_INLINE bool8 raf_bitraf_verify(s32 idx) {")
L("    if (RAF_UNLIKELY(idx < 0 || (u32)idx >= _g_bitraf_count)) return RAF_FALSE;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    if (!RAF_BLOCK_WITNESS_GET(b)) return RAF_FALSE;")
L("    u32 stored = (u32)b->crc_lo[0]")
L("               | ((u32)b->crc_lo[1] <<  8)")
L("               | ((u32)b->crc_lo[2] << 16)")
L("               | ((u32)b->crc_lo[3] << 24);")
L("    u32 calc = raf_crc32c_fast(b->data, 59);")
L("    if (stored != calc) { _g_bitraf_errors++; return RAF_FALSE; }")
L("    return RAF_TRUE;")
L("}")
L()
L("/* Extrai plano nibble HI de um bloco — só executa se Witness=true */")
L("RAF_INLINE void raf_bitraf_nibble_hi_plane(s32 idx, u8* out, u32 n) {")
L("    if (!raf_bitraf_verify(idx)) return;  /* INVARIANTE */")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    u32 lim = (n < 59) ? n : 59;")
L("    for (u32 i = 0; i < lim; i++) out[i] = raf_nibble_hi(b->data[i]);")
L("}")
L()
L("/* Extrai plano nibble LO */")
L("RAF_INLINE void raf_bitraf_nibble_lo_plane(s32 idx, u8* out, u32 n) {")
L("    if (!raf_bitraf_verify(idx)) return;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    u32 lim = (n < 59) ? n : 59;")
L("    for (u32 i = 0; i < lim; i++) out[i] = raf_nibble_lo(b->data[i]);")
L("}")
L()
L("/* XOR entre dois blocos — re-sela dst após modificação */")
L("RAF_NOINLINE void raf_bitraf_xor_blocks(s32 dst, s32 src) {")
L("    if (!raf_bitraf_verify(dst)) return;")
L("    if (!raf_bitraf_verify(src)) return;")
L("    RafBlock* d = &_g_bitraf_blocks[dst];")
L("    const RafBlock* s_ = &_g_bitraf_blocks[src];")
L("#ifdef RAF_HAS_NEON")
L("    /* NEON: XOR 48 bytes em 3 registros de 16 bytes */")
L("    const u8* sp = s_->data;")
L("    u8* dp = d->data;")
L("    __asm__ __volatile__(")
L('        "ld1 {v0.16b, v1.16b, v2.16b}, [%1]\\n\\t"')
L('        "ld1 {v3.16b, v4.16b, v5.16b}, [%0]\\n\\t"')
L('        "eor v3.16b, v3.16b, v0.16b\\n\\t"')
L('        "eor v4.16b, v4.16b, v1.16b\\n\\t"')
L('        "eor v5.16b, v5.16b, v2.16b\\n\\t"')
L('        "st1 {v3.16b, v4.16b, v5.16b}, [%0]\\n\\t"')
L("        : \"+r\"(dp) : \"r\"(sp)")
L("        : \"memory\",\"v0\",\"v1\",\"v2\",\"v3\",\"v4\",\"v5\"")
L("    );")
L("    /* últimos 11 bytes escalares */")
L("    for (u32 i = 48; i < 59; i++) d->data[i] ^= s_->data[i];")
L("#else")
L("    for (u32 i = 0; i < 59; i++) d->data[i] ^= s_->data[i];")
L("#endif")
L("    raf_bitraf_seal(dst);")
L("}")
L()
L("/* AND entre dois blocos */")
L("RAF_NOINLINE void raf_bitraf_and_blocks(s32 dst, s32 src) {")
L("    if (!raf_bitraf_verify(dst)) return;")
L("    if (!raf_bitraf_verify(src)) return;")
L("    RafBlock* d = &_g_bitraf_blocks[dst];")
L("    const RafBlock* s_ = &_g_bitraf_blocks[src];")
L("    for (u32 i = 0; i < 59; i++) d->data[i] &= s_->data[i];")
L("    raf_bitraf_seal(dst);")
L("}")
L()
L("/* OR entre dois blocos */")
L("RAF_NOINLINE void raf_bitraf_or_blocks(s32 dst, s32 src) {")
L("    if (!raf_bitraf_verify(dst)) return;")
L("    if (!raf_bitraf_verify(src)) return;")
L("    RafBlock* d = &_g_bitraf_blocks[dst];")
L("    const RafBlock* s_ = &_g_bitraf_blocks[src];")
L("    for (u32 i = 0; i < 59; i++) d->data[i] |= s_->data[i];")
L("    raf_bitraf_seal(dst);")
L("}")
L()
L("/* NOT de um bloco */")
L("RAF_NOINLINE void raf_bitraf_not_block(s32 idx) {")
L("    if (!raf_bitraf_verify(idx)) return;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    for (u32 i = 0; i < 59; i++) b->data[i] = ~b->data[i];")
L("    raf_bitraf_seal(idx);")
L("}")
L()
L("/* Preenche bloco com padrão de nibble HI=hi, LO=lo */")
L("RAF_INLINE void raf_bitraf_fill_nibble(s32 idx, u8 hi, u8 lo) {")
L("    if (idx < 0 || (u32)idx >= _g_bitraf_count) return;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    u8 pat = raf_nibble_pack(hi, lo);")
L("    raf_memset(b->data, pat, 59);")
L("    raf_bitraf_seal(idx);")
L("}")
L()
L("/* Escreve u64 em um bloco (big-endian, a partir do byte offset) */")
L("RAF_INLINE void raf_bitraf_write_u64(s32 idx, u32 offset, u64 val) {")
L("    if (idx < 0 || (u32)idx >= _g_bitraf_count) return;")
L("    if (offset + 8 > 59) return;")
L("    RafBlock* b = &_g_bitraf_blocks[idx];")
L("    b->data[offset+0] = (u8)(val >> 56);")
L("    b->data[offset+1] = (u8)(val >> 48);")
L("    b->data[offset+2] = (u8)(val >> 40);")
L("    b->data[offset+3] = (u8)(val >> 32);")
L("    b->data[offset+4] = (u8)(val >> 24);")
L("    b->data[offset+5] = (u8)(val >> 16);")
L("    b->data[offset+6] = (u8)(val >>  8);")
L("    b->data[offset+7] = (u8)(val >>  0);")
L("    RAF_BLOCK_DIRTY_SET(b);")
L("    RAF_BLOCK_WITNESS_CLR(b);  /* invalida até re-seal */")
L("}")
L()

# SEÇÃO 9 — ZIPRAF
L("// " + "=" * 117)
L("// SEÇÃO 9 — ZIPRAF: COMPRESSÃO POR OVERLAY DE MESMA GEOMETRIA")
L("// Ref: llamaRafaelia/rafaelia-baremetal/src/zipraf.c")
L("//      llamaRafaelia/SPEC.md §ZIPRAF: overlay de mesma geometria")
L("//      termux-app-rafacodephi/RAFAELIA_METHODOLOGY.md")
L("// Propósito: RLE por nibble + XOR overlay — sem malloc")
L("//            Buffer de destino fornecido pelo chamador (stack ou arena)")
L("// " + "=" * 117)
L()
L("/* Comprimir: RLE de nibble LO (run até 15 repetições por entry) */")
L("RAF_NOINLINE usize raf_zipraf_compress(const u8* src, usize src_len,")
L("                                        u8* dst, usize dst_cap) {")
L("    usize wi = 0, ri = 0;")
L("    while (ri < src_len && wi < dst_cap) {")
L("        u8 val = raf_nibble_lo(src[ri]);")
L("        u8 cnt = 1;")
L("        while (ri+cnt < src_len && cnt < 15 &&")
L("               raf_nibble_lo(src[ri+cnt]) == val) cnt++;")
L("        dst[wi++] = raf_nibble_pack(cnt, val);")
L("        ri += cnt;")
L("    }")
L("    return wi;")
L("}")
L()
L("/* Descomprimir */")
L("RAF_NOINLINE usize raf_zipraf_decompress(const u8* src, usize src_len,")
L("                                          u8* dst, usize dst_cap) {")
L("    usize wi = 0;")
L("    for (usize i = 0; i < src_len && wi < dst_cap; i++) {")
L("        u8 cnt = raf_nibble_hi(src[i]);")
L("        u8 val = raf_nibble_lo(src[i]);")
L("        for (u8 j = 0; j < cnt && wi < dst_cap; j++) dst[wi++] = val;")
L("    }")
L("    return wi;")
L("}")
L()
L("/* Compressão de bloco BITRAF com ZIPRAF — resultado em arena */")
L("RAF_INLINE usize raf_zipraf_compress_block(s32 idx, u8* dst, usize dst_cap) {")
L("    if (!raf_bitraf_verify(idx)) return 0;")
L("    return raf_zipraf_compress(_g_bitraf_blocks[idx].data, 59, dst, dst_cap);")
L("}")
L()

# SEÇÃO 10 — RAFSTORE
L("// " + "=" * 117)
L("// SEÇÃO 10 — RAFSTORE: KV+RING+CACHE BINARY (sem malloc, sem heap)")
L("// Ref: termux-app-rafacodephi/RAFCODEPHI_V3_NOTES.md §V3.3 RafStore + IO/footprint")
L("//      termux-app-rafacodephi/RAFCODEPHI_V3_NOTES.md §rafstore: .rafstore/fileaudit.bin")

L("// Propósito: cache binário stat-only para fileaudit (evita re-reads)")
L("//            ring buffer para IPC baremetal, KV store para configuração")
L("// " + "=" * 117)
L()
L("/* ---- Ring Buffer ---- */")
L("typedef struct {")
L("    u64 slots[RAF_RING_CAP];")
L("    u32 head;")
L("    u32 tail;")
L("    u32 count;")
L("    u32 drops;  /* contagem de drops por overflow */")
L("    u8  _pad[8];")
L("} RafRing;")
L()
L("RAF_INLINE void  raf_ring_init(RafRing* r)  { raf_memset(r, 0, sizeof(*r)); }")
L("RAF_INLINE bool8 raf_ring_full(RafRing* r)  { return r->count == RAF_RING_CAP; }")
L("RAF_INLINE bool8 raf_ring_empty(RafRing* r) { return r->count == 0; }")
L()
L("RAF_INLINE bool8 raf_ring_push(RafRing* r, u64 v) {")
L("    if (raf_ring_full(r)) { r->drops++; return RAF_FALSE; }")
L("    r->slots[r->tail] = v;")
L("    r->tail = (r->tail + 1) % RAF_RING_CAP;")
L("    r->count++;")
L("    return RAF_TRUE;")
L("}")
L()
L("RAF_INLINE bool8 raf_ring_pop(RafRing* r, u64* out) {")

# SEÇÃO 12 — CICLO RAFAELIA ψ→χ→ρ→∆→Σ→Ω
L("// " + "=" * 117)
L("// SEÇÃO 12 — CICLO RAFAELIA ψ→χ→ρ→∆→Σ→Ω (engine principal)")
L("// Ref: llamaRafaelia/rafaelia-baremetal/ (ciclo descrito no README)")

L("/* Vetor ético de 13 dimensões — Ref: llamaRafaelia README */")
L("typedef struct {")
L("    u8 intencao;           /* 0-255: nível de intenção declarada */")

L("    u8 confusao;           /* nível de confusão (ruim se alto) */")
L("    u8 risco_vida;         /* risco (ruim se alto) */")

L("    u8 dano_irreversivel;  /* (ruim se alto) */")

L("/* Limiares éticos conservadores */")
L("#define RAF_ETH_MIN_CUIDADO       64")
L("#define RAF_ETH_MIN_CERTEZA       32")
L("#define RAF_ETH_MAX_RISCO         128")
L("#define RAF_ETH_MAX_CONFUSAO      200")
L("#define RAF_ETH_MAX_DANO_IRR      64")
L("#define RAF_ETH_MAX_QUEBRA_CONF   128")

L("RAF_PURE bool8 raf_ethica_should_proceed(const RafEthicaVec* v) {")
L("    if (v->risco_vida        > RAF_ETH_MAX_RISCO)       return RAF_FALSE;")
L("    if (v->confusao          > RAF_ETH_MAX_CONFUSAO)    return RAF_FALSE;")
L("    if (v->dano_irreversivel > RAF_ETH_MAX_DANO_IRR)    return RAF_FALSE;")

L("    if (!v->nao_ferir)                                  return RAF_FALSE;")
L("    return RAF_TRUE;")
L("}")
L()
L("typedef struct {")
L("    u64  psi;      /* ψ — intenção (hash da entrada) */")
L("    u64  chi;      /* χ — observação (vetor de features) */")
L("    u64  rho;      /* ρ — estado de ruído/entropia */")
L("    u64  delta;    /* ∆ — transmutação (resultado da regra) */")
L("    u64  sigma;    /* Σ — memória coerente (acumulador) */")
L("    u64  omega;    /* Ω — estado final aprovado pelo filtro ético */")
L("    u64  sigma_prev;/* Σ anterior — para detecção de regressão */")
L("    u32  seq;")
L("    u32  ethica_blocks;   /* contador de bloqueios éticos */")
L("    u8   valid;")
L("    u8   ethica_ok;")
L("    u8   _pad[6];")
L("} RafCycleState;")
L()
L("/* Fibonacci-Rafael: sequência para ρ — inteiro Q32, sem float */")
L("#define RAF_Q32_SQRT3_HALF  0x6ED9EBA1u")
L("#define RAF_Q32_PI_SIN999   0x0A3D70A4u")
L()
L("RAF_INLINE u64 raf_fib_rafael(u32 n) {")
L("    u64 f = 0x1000000000000000ULL;")
L("    for (u32 i = 0; i < n; i++) {")
L("        f = (f >> 32) * (u64)RAF_Q32_SQRT3_HALF;")
L("        f += (u64)RAF_Q32_PI_SIN999;")
L("    }")
L("    return f;")
L("}")
L()
L("RAF_NOINLINE bool8 raf_cycle_run(RafCycleState* cs, const u8* input, usize ilen,")
L("                                  const RafEthicaVec* ev) {")
L("    cs->valid      = 0;")
L("    cs->ethica_ok  = 0;")
L("    cs->sigma_prev = cs->sigma;")
L("    cs->seq++;")
L()
L("    /* ψ — hash da entrada como intenção */")
L("    cs->psi   = raf_hash64_fnv1a(input, ilen);")
L()
L("    /* χ — observação: transforma hash em vetor de features */")
L("    cs->chi   = raf_hash64_mix(cs->psi ^ (u64)cs->seq);")
L()
L("    /* ρ — entropia fibonacci-rafael baseada no seq */")
L("    cs->rho   = raf_fib_rafael(cs->seq & 0xFF);")
L()
L("    /* ∆ — transmutação: XOR dos três estados */")
L("    cs->delta = cs->psi ^ cs->chi ^ cs->rho;")
L()
L("    /* Σ — memória coerente: acumula com rotação por seq */")
L("    u32 rot   = cs->seq & 63u;")
L("    cs->sigma ^= (cs->delta << rot) | (cs->delta >> (64u - rot));")
L()
L("    /* Ω — gate ético obrigatório */")
L("    if (!raf_ethica_should_proceed(ev)) {")
L("        cs->ethica_blocks++;")
L("        RAF_LOG_ERR(\"OMEGA: ethica blocked\", (u64)cs->seq);")
L("        return RAF_FALSE;")
L("    }")
L("    cs->ethica_ok = 1;")
L()
L("    /* Ω aprovado: aplica hash final */")
L("    cs->omega = raf_hash64_mix(cs->sigma ^ cs->delta);")
L("    cs->valid = 1;")
L("    return RAF_TRUE;")
L("}")
L()

L("/* Flip Horizontal — espelha esquerda-direita (3.24x speedup))")
L(" * Ref: BOOSTERS.md §2 Flip Horizontal / IMPLEMENTACAO_BAREMETAL.md §mx_flip_h */")
L("RAF_NOINLINE void mx_flip_h(mx_t* m) {")
L("    for (u32 i = 0; i < m->r; i++) {")

L("        }")
L("    }")

L("        u32 top = i * m->c, bot = (m->r-1-i) * m->c;")
L("        for (u32 j = 0; j < m->c; j++) {")
L("            float t = m->m[top+j]; m->m[top+j] = m->m[bot+j]; m->m[bot+j] = t;")

L("        for (u32 j = i+1; j < m->c; j++) {")
L("            float t = MX_AT(m,i,j); MX_AT(m,i,j) = MX_AT(m,j,i); MX_AT(m,j,i) = t;")

L("/* Multiplicação de matrizes — cache blocking 32 (2.71x speedup)")

L("RAF_NOINLINE bool8 mx_mul(const mx_t* a, const mx_t* b, mx_t* r, RafArena* arena) {")
L("    if (a->c != b->r) return RAF_FALSE;")
L("    if (!mx_init(r, a->r, b->c, arena)) return RAF_FALSE;")
L("#define MX_MUL_BLOCK 32")
L("    for (u32 i = 0; i < a->r; i += MX_MUL_BLOCK) {")
L("        for (u32 j = 0; j < b->c; j += MX_MUL_BLOCK) {")
L("            for (u32 k = 0; k < a->c; k += MX_MUL_BLOCK) {")
L("                u32 imax = (i+MX_MUL_BLOCK<a->r)?i+MX_MUL_BLOCK:a->r;")
L("                u32 jmax = (j+MX_MUL_BLOCK<b->c)?j+MX_MUL_BLOCK:b->c;")
L("                u32 kmax = (k+MX_MUL_BLOCK<a->c)?k+MX_MUL_BLOCK:a->c;")
L("                for (u32 ii=i; ii<imax; ii++) {")
L("                    for (u32 jj=j; jj<jmax; jj++) {")
L("                        float s = 0.0f;")
L("                        for (u32 kk=k; kk<kmax; kk++)")
L("                            s += MX_AT(a,ii,kk) * MX_AT(b,kk,jj);")
L("                        MX_AT(r,ii,jj) += s;")
L("                    }")
L("                }")
L("            }")
L("        }")
L("    }")
L("    return RAF_TRUE;")
L("}")

L("}")

L("/* Determinante — eliminação gaussiana in-place (2.68x speedup) */")
L("RAF_NOINLINE float mx_det(mx_t* m) {")
L("    if (m->r != m->c) return 0.0f;")
L("    u32 n = m->r;")
L("    float det = 1.0f;")
L("    /* cópia temporária na stack se pequeno */")
L("    if (n > 16) return 0.0f; /* limitado a 16x16 sem heap */")
L("    float tmp[16*16];")
L("    raf_memcpy(tmp, m->m, sizeof(float)*n*n);")
L("    mx_t t; t.m = tmp; t.r = n; t.c = n; t.cap = n*n;")
L("    for (u32 i = 0; i < n; i++) {")
L("        /* pivotamento parcial */")
L("        u32 pivot = i;")
L("        for (u32 k=i+1; k<n; k++)")
L("            if (fm_abs(MX_AT(&t,k,i)) > fm_abs(MX_AT(&t,pivot,i))) pivot = k;")
L("        if (pivot != i) {")
L("            for (u32 j=0; j<n; j++) {")
L("                float tmp2 = MX_AT(&t,i,j);")
L("                MX_AT(&t,i,j) = MX_AT(&t,pivot,j);")
L("                MX_AT(&t,pivot,j) = tmp2;")
L("            }")
L("            det = -det;")
L("        }")
L("        if (MX_AT(&t,i,i) == 0.0f) return 0.0f;")
L("        det *= MX_AT(&t,i,i);")
L("        float inv = 1.0f / MX_AT(&t,i,i);")
L("        for (u32 j=i+1; j<n; j++) {")
L("            float factor = MX_AT(&t,j,i) * inv;")
L("            for (u32 k=i; k<n; k++)")
L("                MX_AT(&t,j,k) -= factor * MX_AT(&t,i,k);")
L("        }")
L("    }")
L("    return det;")
L("}")
L()

# vop_dot
L("/* Produto escalar — ARM64 NEON 4 floats/ciclo")
L(" * Ref: BOOSTERS.md §Assembly ARMv8-A 64-bit dot product")
L(" * bm_dot_neon: ld1 {v1.4s},[x0]; ld1 {v2.4s},[x1]; fmla v0.4s,v1.4s,v2.4s */")
L("RAF_HOT float vop_dot(const float* a, const float* b, u32 n) {")
L("#ifdef RAF_HAS_NEON")
L("    float result_arr[4] __attribute__((aligned(16))) = {0,0,0,0};")
L("    u32 ni = n & ~3u;")
L("    if (ni) {")
L("        __asm__ __volatile__(")
L('            "movi v0.4s, #0\\n\\t"')
L('            "1:\\n\\t"')
L('            "ld1 {v1.4s}, [%0], #16\\n\\t"')
L('            "ld1 {v2.4s}, [%1], #16\\n\\t"')
L('            "fmla v0.4s, v1.4s, v2.4s\\n\\t"')
L('            "subs %2, %2, #4\\n\\t"')
L('            "b.gt 1b\\n\\t"')
L('            "st1 {v0.4s}, [%3]\\n\\t"')
L("            : \"+r\"(a), \"+r\"(b), \"+r\"(ni)")
L("            : \"r\"(result_arr)")
L("            : \"memory\", \"v0\", \"v1\", \"v2\"")
L("        );")
L("    }")
L("    float sum = result_arr[0]+result_arr[1]+result_arr[2]+result_arr[3];")
L("    for (u32 i = n&~3u; i < n; i++) sum += a[i]*b[i];")
L("    return sum;")
L("#else")
L("    float sum = 0.0f;")
L("    for (u32 i = 0; i < n; i++) sum += a[i]*b[i];")
L("    return sum;")
L("#endif")
L("}")

L("/* Desquantiza bloco Q4_0 em float32 (32 floats) — sem malloc")
L(" * Ref: ggml.c ggml_dequantize_row_q4_0 */")
L("RAF_INLINE void raf_q4_dequantize(const RafQ4Block* blk, float* out) {")
L("    float scale = raf_f16_to_f32(blk->scale_f16);")
L("    for (u32 i = 0; i < 16; i++) {")
L("        s32 q0 = (s32)(blk->quants[i] & 0xFu) - 8;  /* LO nibble */")
L("        s32 q1 = (s32)(blk->quants[i] >> 4)   - 8;  /* HI nibble */")

L("        out[i*2+1] = (float)q1 * scale;")
L("    }")
L("}")

L("/* Quantiza 32 floats em bloco Q4_0 — sem malloc */")
L("RAF_NOINLINE void raf_q4_quantize(const float* in, RafQ4Block* out) {")
L("    float max_v = 0.0f;")
L("    for (u32 i = 0; i < 32; i++) {")
L("        float av = fm_abs(in[i]);")
L("        if (av > max_v) max_v = av;")
L("    }")
L("    float scale = max_v / 7.0f;")
L("    float inv_scale = (scale != 0.0f) ? (1.0f / scale) : 0.0f;")
L("    out->scale_f16 = raf_f32_to_f16(scale);")
L("    for (u32 i = 0; i < 16; i++) {")
L("        s32 q0 = (s32)(in[i*2]   * inv_scale + 8.5f);")
L("        s32 q1 = (s32)(in[i*2+1] * inv_scale + 8.5f);")
L("        q0 = (q0 < 0)?0:(q0>15)?15:q0;")
L("        q1 = (q1 < 0)?0:(q1>15)?15:q1;")
L("        out->quants[i] = (u8)((q1 << 4) | q0);")
L("    }")
L("}")

L("/* Dot product Q4_0 × float32 — ARM64 NEON inline")
L(" * Ref: llamaRafaelia/assembler/ ggml_vec_dot_q4_0_q8_0 ARM64")
L(" * Ref: BOOSTERS.md §NEON ARMv8-A — fmla v0.4s, v1.4s, v2.4s")
L(" * Speedup: 3.5x vs escalar (BOOSTERS.md §Benchmark NEON) */")
L("RAF_HOT float raf_q4_dot_f32(const RafQ4Block* blks, u32 n_blocks,")
L("                              const float* vec) {")
L("    float sum = 0.0f;")
L("    for (u32 b = 0; b < n_blocks; b++) {")
L("        float scale = raf_f16_to_f32(blks[b].scale_f16);")
L("        const u8* q = blks[b].quants;")
L("        const float* v = vec + b * 32;")
L("#ifdef RAF_HAS_NEON")
L("        /* Desquantiza 16 bytes → 32 floats e faz dot via NEON */")
L("        float dq[32] __attribute__((aligned(16)));")
L("        for (u32 i = 0; i < 16; i++) {")
L("            dq[i*2]   = (float)((s32)(q[i] & 0xFu) - 8) * scale;")
L("            dq[i*2+1] = (float)((s32)(q[i] >> 4)   - 8) * scale;")

L("        sum += vop_dot(dq, v, 32);")
L("#else")
L("        for (u32 i = 0; i < 16; i++) {")
L("            sum += (float)((s32)(q[i]&0xFu)-8)*scale*v[i*2];")
L("            sum += (float)((s32)(q[i]>>4) -8)*scale*v[i*2+1];")
L("        }")
L("#endif")
L("    }")
L("    return sum;")
L("}")

L("/* Layer norm sem malloc — in-place, epsilon=1e-5 */")
L("RAF_NOINLINE void raf_layer_norm(float* v, u32 n, const float* gamma, const float* beta) {")
L("    float mean = 0.0f;")
L("    for (u32 i=0; i<n; i++) mean += v[i];")
L("    mean /= (float)n;")
L("    float var = 0.0f;")
L("    for (u32 i=0; i<n; i++) { float d = v[i]-mean; var += d*d; }")
L("    var /= (float)n;")
L("    float inv = fm_rsqrt(var + 1e-5f);")
L("    for (u32 i=0; i<n; i++) {")
L("        v[i] = (v[i]-mean)*inv;")
L("        if (gamma) v[i] *= gamma[i];")
L("        if (beta)  v[i] += beta[i];")
L("    }")
L("}")
L()
L("/* RMS Norm — sem malloc, sem heap */")
L("RAF_NOINLINE void raf_rms_norm(float* v, u32 n, float eps) {")
L("    float ss = vop_dot(v, v, n) / (float)n + eps;")
L("    float inv = fm_rsqrt(ss);")
L("    vop_scale(v, inv, n);")
L("}")

L("/* Audita arquivo: stat-only se em cache (rafstore), re-read se mudou */")
L("RAF_NOINLINE bool8 raf_fileaudit_entry(RafStore* cache, const char* path) {")
L("    u64 path_hash = raf_hash64_fnv1a((const u8*)path, raf_strlen(path));")
L("    RafStat st;")
L("    if (raf_stat(path, &st) < 0) return RAF_FALSE;")
L("    /* Verifica cache — stat-only se não mudou */")
L("    RafStoreEntry cached;")
L("    if (raf_rafstore_get(cache, path_hash, &cached)) {")
L("        if (cached.size == st.size && cached.mtime == st.mtime_sec) {")
L("            /* Cache hit: stat-only, sem read extra — zero I/O adicional */")
L("            if (_g_fileaudit_count < RAF_FILEAUDIT_MAX) {")
L("                RafFileAuditEntry* e = &_g_fileaudit[_g_fileaudit_count++];")
L("                raf_strcpy(e->path, path, 128);")
L("                e->path_hash = path_hash;")
L("                e->size      = st.size;")
L("                e->mtime     = st.mtime_sec;")
L("                e->crc       = cached.crc;")
L("                e->flags     = RAF_FILEAUDIT_FLAG_AUDITED | RAF_FILEAUDIT_FLAG_KNOWN;")
L("                return RAF_TRUE;")
L("            }")
L("            return RAF_FALSE;")
L("        }")
L("        /* Arquivo mudou — marca como changed */")
L("    }")
L("    /* Cache miss ou mudança: lê e calcula CRC */")
L("    s32 fd = raf_open(path, 0 /* O_RDONLY */, 0);")
L("    u32 crc = 0;")
L("    if (fd >= 0) {")
L("        u8 buf[4096];")
L("        crc = 0xFFFFFFFFu;")
L("        ssize nr;")
L("        while ((nr = raf_read(fd, buf, sizeof(buf))) > 0)")
L("            crc = raf_crc32c_fast(buf, (usize)nr) ^ crc; /* acumula */")
L("        crc ^= 0xFFFFFFFFu;")
L("        raf_close(fd);")
L("    }")
L("    raf_rafstore_put(cache, path_hash, (u64)st.size, (u64)st.mtime_sec, crc);")
L("    if (_g_fileaudit_count < RAF_FILEAUDIT_MAX) {")
L("        RafFileAuditEntry* e = &_g_fileaudit[_g_fileaudit_count++];")
L("        raf_strcpy(e->path, path, 128);")
L("        e->path_hash = path_hash; e->size = st.size;")
L("        e->mtime = st.mtime_sec; e->crc = crc;")
L("        e->flags = RAF_FILEAUDIT_FLAG_AUDITED;")
L("    }")
L("    return RAF_TRUE;")
L("}")
