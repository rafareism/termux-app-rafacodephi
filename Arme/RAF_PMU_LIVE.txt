/* Domínio público: definições do kernel Linux (GPL-2.0+) */
struct pea_t {
    uint32_t type;
    uint32_t size;
    uint64_t config;
    uint64_t sample_period;
    uint64_t sample_type;
    uint64_t read_format;

    int32_t  wakeup_events;
    uint32_t bp_type;
    uint64_t bp_addr;
    uint64_t bp_len;
} __attribute__((packed));

#define PERF_TYPE_HARDWARE      0
#define PERF_TYPE_SOFTWARE      1
#define PERF_COUNT_HW_CYCLES    0
#define PERF_COUNT_HW_INSTR     1
#define PERF_COUNT_HW_CACHE_REF 2
#define PERF_COUNT_HW_CACHE_MISS 3
#define PERF_COUNT_HW_BRANCH_MISS 5
#define PERF_COUNT_HW_STALL_FRONT 6
#define PERF_COUNT_SW_TASK_CLOCK 1  /* fallback: sempre disponível */

#define SYS_PERF_EVENT_OPEN 364
#else
#define SYS_PERF_EVENT_OPEN 241  /* arm64 */
#endif

#define IOCTL_ENABLE 0x00002400  /* PERF_EVENT_IOC_ENABLE */

/* Abre um contador PMU. Retorna fd ou -1 */
static int pmu_open(uint32_t type, uint64_t config) {
    struct pea_t attr;
    memset(&attr, 0, sizeof(attr));
    attr.type   = type;
    attr.size   = sizeof(attr);
    attr.config = config;
    attr.flags  = 1ULL;  /* disabled=1 */
    int fd = (int)syscall(SYS_PERF_EVENT_OPEN, &attr,
                          0, -1, -1, 0);
    return fd;
}

/* Lê contador */
static uint64_t pmu_read(int fd) {
    uint64_t v = 0;
    if (fd < 0) return 0;
    read(fd, &v, 8);
    return v;
}

/* Contexto PMU: até 6 contadores */
typedef struct {
    int fd[6];      /* file descriptors */
    int n_avail;    /* quantos abriram com sucesso */
    int hw_avail;   /* 1 se hardware PMU disponível */
} pmu_ctx_t;

static void pmu_init(pmu_ctx_t *ctx) {
    memset(ctx, 0xff, sizeof(*ctx)); /* -1 em todos */
    ctx->n_avail = 0;
    ctx->hw_avail = 0;

    /* tenta hardware primeiro */
    static const uint64_t HW_EVENTS[] = {
        PERF_COUNT_HW_CYCLES, PERF_COUNT_HW_INSTR,
        PERF_COUNT_HW_CACHE_REF, PERF_COUNT_HW_CACHE_MISS,
        PERF_COUNT_HW_BRANCH_MISS, PERF_COUNT_HW_STALL_FRONT,
    };
    for (int i = 0; i < 6; i++) {
        ctx->fd[i] = pmu_open(PERF_TYPE_HARDWARE, HW_EVENTS[i]);
        if (ctx->fd[i] >= 0) { ctx->n_avail++; ctx->hw_avail = 1; }
    }

    /* fallback: task_clock (software, sempre disponível) */
    if (!ctx->hw_avail) {
        ctx->fd[0] = pmu_open(PERF_TYPE_SOFTWARE, PERF_COUNT_SW_TASK_CLOCK);
        if (ctx->fd[0] >= 0) ctx->n_avail = 1;
    }
}

static void pmu_start(pmu_ctx_t *ctx) {
    for (int i = 0; i < 6; i++) {
        if (ctx->fd[i] < 0) continue;
        ioctl(ctx->fd[i], IOCTL_RESET,  0);
        ioctl(ctx->fd[i], IOCTL_ENABLE, 0);
    }
}

static void pmu_stop(pmu_ctx_t *ctx, uint64_t *out) {
    for (int i = 0; i < 6; i++) {
        out[i] = (ctx->fd[i] >= 0) ? pmu_read(ctx->fd[i]) : 0;
        if (ctx->fd[i] >= 0) ioctl(ctx->fd[i], 0x00002401, 0); /* DISABLE */
    }
}

static void pmu_close(pmu_ctx_t *ctx) {
    for (int i = 0; i < 6; i++)
        if (ctx->fd[i] >= 0) close(ctx->fd[i]);
}

/* Workload: EMA Q16.16 × N (registradores puros, sem variável nomeada) */
static uint32_t workload_ema(uint32_t n) {
    register uint32_t C = 32768u;
    register uint32_t H = 32768u;
    /* r4=SPIRAL seed para evitar constante inline */
    register uint32_t s = 56755u;
    for (uint32_t i = 0; i < n; i++) {
        /* C = (C×49152 + s×16384) >> 16 */
        C = (uint32_t)(((uint64_t)C*49152u + (uint64_t)s*16384u) >> 16);
        H = (uint32_t)(((uint64_t)H*49152u + (uint64_t)(65535u-s)*16384u) >> 16);
        s = (uint32_t)(((uint64_t)s*56755u) & 0xFFFFu);
    }
    return C ^ H;  /* usa resultado para evitar dead code elim */
}

/* Workload: memory latency pointer chase (mede miss rate real) */
static uint32_t workload_chase(uint8_t *buf, uint32_t sz) {
    /* cria lista encadeada com stride aleatório via XOR shift */
    uint32_t *ptrs = (uint32_t*)buf;
    uint32_t n = sz / 4;
    /* init: ptr[i] = próximo índice (shuffle) */
    for (uint32_t i = 0; i < n; i++) ptrs[i] = (i + 1) % n;
    /* shuffle via XOR-shift */
    uint32_t seed = 0x9E3779B9u;
    for (uint32_t i = n-1; i > 0; i--) {
        seed ^= seed << 13; seed ^= seed >> 17; seed ^= seed << 5;
        uint32_t j = seed % (i+1);
        uint32_t t = ptrs[i]; ptrs[i] = ptrs[j]; ptrs[j] = t;
    }
    /* chase: percorre a lista — força cache miss */
    uint32_t idx = 0;
    for (uint32_t i = 0; i < n && i < 1024; i++)
        idx = ptrs[idx];
    return idx;
}

/**
 * raf_cache_detector.c — Auto Cache Boundary Detector ARM32
 * SPDX-License-Identifier: GPL-3.0-only
 *
 * Algoritmo:
 *   1. varre 1KB→4MB em passos multiplicativos (×1.5)
 *   2. mede bandwidth: MB/s = (bytes × 1e6) / wall_us
 *   3. detecta quebras: queda > 25% = transição de camada
 *   4. classifica: L1_SIZE, L2_SIZE baseado nas quebras
 *
 * Resultado: mapa completo da hierarquia de cache do SoC
 * sem PMU, sem root, só timing + NEON.
 *
 * Seus números medidos (Helio G25):
 *   8KB  → 11482 MB/s  ← L1 peak
 *   64KB → 6261 MB/s   ← L2 início (queda ~45%)
 *   512KB→ 3090 MB/s   ← L2/RAM transição
 *   1MB+ → ~1200 MB/s  ← RAM puro
 */
#define _POSIX_C_SOURCE 200809L
#include <stdint.h>
#include <stddef.h>
#include <string.h>
#include <unistd.h>
#include <time.h>
#ifdef __ARM_NEON
#include <arm_neon.h>
#endif

#define ARENA_SZ (5u*1024u*1024u)
static uint8_t __attribute__((aligned(64))) _arena[ARENA_SZ];
static uint32_t _bump=0;
static void*A(uint32_t n,uint32_t al){
    uint32_t m=al-1,s=(_bump+m)&~m,e=s+n;
    if(e>ARENA_SZ)return 0;_bump=e;return _arena+s;
}

static uint64_t now_us(){
    struct timespec t;
    clock_gettime(CLOCK_MONOTONIC,&t);
    return(uint64_t)t.tv_sec*1000000u+(uint64_t)t.tv_nsec/1000u;
}

/* Leitura sequencial NEON: 16 bytes/iteração
 * SEM variável nomeada: q0..q3 são os dados
 * r4=ptr r5=end — registradores ARM32
 * Compila para: vld1.32 {q0},[r4]! vld1.32 {q1},[r4]! ... */
static uint32_t seq_read_neon(uint8_t *buf, uint32_t sz, uint32_t reps) {
    uint32_t acc = 0;
    for (uint32_t r = 0; r < reps; r++) {
#ifdef __ARM_NEON
        uint8_t *p = buf;
        uint8_t *end = buf + sz;
        uint32x4_t vacc = vdupq_n_u32(0);
        while (p + 64 <= end) {
            uint32x4_t v0 = vld1q_u32((uint32_t*)p);      p+=16;
            uint32x4_t v1 = vld1q_u32((uint32_t*)p);      p+=16;
            uint32x4_t v2 = vld1q_u32((uint32_t*)p);      p+=16;
            uint32x4_t v3 = vld1q_u32((uint32_t*)p);      p+=16;
            vacc = vaddq_u32(vacc, vaddq_u32(vaddq_u32(v0,v1),vaddq_u32(v2,v3)));
        }
        acc ^= vaddvq_u32(vacc);
#else
        uint32_t *p = (uint32_t*)buf;
        for (uint32_t i = 0; i < sz/4; i++) acc ^= p[i];
#endif
    }
    return acc;
}

/* Mede bandwidth em MB/s para tamanho sz */
static uint32_t measure_bw(uint8_t *buf, uint32_t sz) {
    /* calibra repetições para ter pelo menos 10ms de medição */
    uint32_t reps = 1;
    if (sz <  8192) reps = 500;
    else if (sz < 65536)  reps = 100;
    else if (sz < 524288) reps = 20;
    else reps = 5;

    uint64_t t0 = now_us();
    uint32_t r  = seq_read_neon(buf, sz, reps);
    uint64_t us = now_us() - t0;
    if (!us) us = 1;
    (void)r;

    uint64_t bytes = (uint64_t)sz * reps;
    /* MB/s = bytes / us = bytes × 1e6 / (us × 1e6) = bytes/us */
    return (uint32_t)(bytes / us);  /* MB/s */
}

/* Detecta transição: queda > threshold% */
#define TRANSITION_THRESH 20  /* 20% de queda = nova camada */

typedef struct {
    uint32_t size_kb;
    uint32_t bw_mbs;
    uint8_t  layer;       /* 0=L1 1=L2 2=L3/LLC 3=RAM */
    uint8_t  transition;  /* 1 se é ponto de transição */
} sample_t;

static const char HX[]="0123456789ABCDEF";

static void wn(){write(1,"\n",1);}
static void wu(uint32_t v){char b[12];int i=11;b[i]=0;if(!v){b[--i]='0';}else while(v){b[--i]=(char)('0'+v%10);v/=10;}ws(b+i);}

/* Barra ASCII proporcional */
static void bar(uint32_t bw, uint32_t max_bw) {
    uint32_t w = (bw * 30) / (max_bw > 0 ? max_bw : 1);
    if (w > 30) w = 30;
    for (uint32_t i = 0; i < w; i++) write(1, "█", 3); /* UTF-8 block */
    for (uint32_t i = w; i < 30; i++) write(1, "░", 3);
}

int main(void) {
    ws("=== AUTO CACHE BOUNDARY DETECTOR ===\n");
    ws("Alvo: Helio G25 Cortex-A53 ARM32\n\n");

    /* tamanhos: 1KB → 4MB em passos ×2 */
    static const uint32_t SIZES[] = {
        1024, 2048, 4096, 8192, 16384, 32768,
        65536, 131072, 262144, 524288,
        1048576, 2097152, 4194304, 0
    };
    static const char *SIZE_STR[] = {
        "1KB","2KB","4KB","8KB","16KB","32KB",
        "64KB","128KB","256KB","512KB",
        "1MB","2MB","4MB",NULL
    };

    /* aloca buffer máximo */
    uint8_t *buf = (uint8_t*)A(4194304u, 64);
    if (!buf) { ws("OOM\n"); return 1; }

    /* init buffer com padrão SPIRAL (evita zero-page optimization) */
    for (uint32_t i = 0; i < 4194304u; i++)
        buf[i] = (uint8_t)((56755u * (i+1u)) >> 8);

    sample_t samples[13];
    int n_samples = 0;
    uint32_t max_bw = 0;

    /* mede cada tamanho */
    for (int i = 0; SIZES[i]; i++) {
        uint32_t bw = measure_bw(buf, SIZES[i]);
        samples[n_samples].size_kb  = SIZES[i] / 1024;
        samples[n_samples].bw_mbs  = bw;
        samples[n_samples].transition = 0;
        samples[n_samples].layer = 0;
        if (bw > max_bw) max_bw = bw;
        n_samples++;
    }

    /* detecta transições e classifica camadas */
    uint8_t cur_layer = 0;
    uint32_t layer_bw[4] = {0,0,0,0};
    uint32_t layer_size[4] = {0,0,0,0};

    for (int i = 1; i < n_samples; i++) {
        uint32_t prev = samples[i-1].bw_mbs;
        uint32_t curr = samples[i].bw_mbs;
        /* queda > TRANSITION_THRESH% */
        if (prev > 0 && (prev - curr) * 100 / prev > TRANSITION_THRESH
            && curr < prev) {
            samples[i].transition = 1;
            if (cur_layer < 3) cur_layer++;
        }
        samples[i].layer = cur_layer;
        if (!layer_bw[cur_layer]) {
            layer_bw[cur_layer]  = curr;
            layer_size[cur_layer]= samples[i].size_kb;
        }
    }
    samples[0].layer = 0;
    layer_bw[0]  = samples[0].bw_mbs;
    layer_size[0]= samples[0].size_kb;

    /* imprime tabela */
    ws("Size    BW(MB/s)  Camada  Gráfico\n");
    ws("──────  ────────  ──────  ──────────────────────────────\n");
    static const char *LN[]={"L1","L2","LLC","RAM"};
    for (int i = 0; i < n_samples; i++) {
        ws(SIZE_STR[i]); ws("    ");
        wu(samples[i].bw_mbs); ws("        ");
        ws(LN[samples[i].layer]);
        if (samples[i].transition) ws("← BREAK");
        ws("  ");
        bar(samples[i].bw_mbs, max_bw);
        wn();
    }

    /* resultado das camadas detectadas */
    ws("\n=== HIERARQUIA DETECTADA ===\n");
    static const char *LAYER_NAME[]={"L1 Cache","L2 Cache","LLC/L3","RAM/DDR"};
    for (int l = 0; l <= (int)cur_layer; l++) {
        ws(LAYER_NAME[l]); ws(": bw="); wu(layer_bw[l]);

/* NEON hot loop: satura vmla × N_ITER
 * Retorna: soma dos acumuladores (previne dead code elim)
 * HOT PATH: sem variável nomeada — só q0..q7 e r4 */
#define N_ITER 1000000u
static float neon_saturate(void) {
#ifdef __ARM_NEON
    /* 8 acumuladores q0..q7: 8×4 floats = 32 floats em voo */
    float32x4_t a0=vdupq_n_f32(1.0001f), a1=vdupq_n_f32(1.0002f);
    float32x4_t a2=vdupq_n_f32(1.0003f), a3=vdupq_n_f32(1.0004f);
    float32x4_t a4=vdupq_n_f32(0.9999f), a5=vdupq_n_f32(0.9998f);
    float32x4_t a6=vdupq_n_f32(0.9997f), a7=vdupq_n_f32(0.9996f);
    float32x4_t c0=vdupq_n_f32(56755.0f/65536.0f); /* SPIRAL */
    float32x4_t c1=vdupq_n_f32(1.0f-56755.0f/65536.0f);

    for (uint32_t i = 0; i < N_ITER; i++) {
        /* 8 vmlaq em sequência → satura pipeline NEON */
        a0 = vmlaq_f32(a0, c0, a1);
        a1 = vmlaq_f32(a1, c1, a2);
        a2 = vmlaq_f32(a2, c0, a3);
        a3 = vmlaq_f32(a3, c1, a4);
        a4 = vmlaq_f32(a4, c0, a5);
        a5 = vmlaq_f32(a5, c1, a6);
        a6 = vmlaq_f32(a6, c0, a7);
        a7 = vmlaq_f32(a7, c1, a0);
    }
    /* reduz para evitar dead code elimination */
    float32x4_t s=vaddq_f32(vaddq_f32(a0,a1),vaddq_f32(a2,a3));
    s=vaddq_f32(s,vaddq_f32(vaddq_f32(a4,a5),vaddq_f32(a6,a7)));
    float r[4]; vst1q_f32(r,s);
    return r[0]+r[1]+r[2]+r[3];
#else
    /* fallback escalar */
    float acc=0.5f;
    for(uint32_t i=0;i<N_ITER;i++) acc=acc*0.8660f+0.1340f;
    return acc;
#endif
}

/* Motor EMA com chunk adaptativo */
static uint32_t motor_ema_chunked(uint32_t n_iter, uint32_t chunk){
    uint32_t C=32768u,H=32768u,chain=0;
    uint32_t step=0;
    while(step<n_iter){
        uint32_t end=step+chunk; if(end>n_iter)end=n_iter;
        for(uint32_t cy=step;cy<end;cy++){
            uint32_t sv=(uint32_t)(((uint64_t)56755u*(cy+1u))>>8&0xFFFFu);
            C=(uint32_t)(((uint64_t)C*49152+(uint64_t)sv*16384)>>16);
            H=(uint32_t)(((uint64_t)H*49152+(uint64_t)(65535u-sv)*16384)>>16);
            uint32_t phi=(uint32_t)(((uint64_t)(65535u-H)*C)>>16);
            uint32_t p[2]={phi,chain};
            /* CRC inline rápido (XOR shift) */
            chain^=p[0]; chain^=(chain<<13); chain^=(chain>>17); chain^=(chain<<5);
        }
        step=end;
    }
    return C^H^chain;
}
